#include <Adafruit_NeoPixel.h>

#define LED_PIN    15    // Chân DIN của LED RGB
#define NUM_LEDS   2    // Số lượng LED nối tiếp

Adafruit_NeoPixel strip(NUM_LEDS, LED_PIN, NEO_GRB + NEO_KHZ800);

void setAllLED(uint32_t color);

void setup() {
  strip.begin();
  strip.show(); // Tắt hết ban đầu
}

void loop() {
  // Ví dụ: sáng đỏ toàn bộ
  setAllLED(strip.Color(255, 0, 0)); // Đỏ
  delay(500);

  // Xanh lá
  setAllLED(strip.Color(0, 255, 0));
  delay(500);

  // Xanh dương
  setAllLED(strip.Color(0, 0, 255));
  delay(500);
}

// Hàm set tất cả LED về cùng màu
void setAllLED(uint32_t color) {
  for (int i = 0; i < NUM_LEDS; i++) {
    strip.setPixelColor(i, color);
  }
  strip.show();
}
