#include <Arduino.h>

#define RELAY_PIN  4  // Chân GPIO nối relay (ví dụ GPIO4)

void setup() {
  pinMode(RELAY_PIN, OUTPUT);
}

void loop() {
  digitalWrite(RELAY_PIN, HIGH);  // Đóng relay
  delay(1000);                    // Chờ 1 giây
  digitalWrite(RELAY_PIN, LOW);   // Mở relay
  delay(1000);                    // Chờ 1 giây
}
